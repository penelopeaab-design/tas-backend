import pytest
from fastapi.testclient import TestClient
import sys
import os
sys.path.insert(0, os.path.abspath(os.path.join(os.path.dirname(__file__), '..')))
from main import app

client = TestClient(app)
FIXTURE_DIR = os.path.join(os.path.dirname(__file__), "fixtures")

def test_health():
    response = client.get("/health")
    assert response.status_code == 200
    assert response.json() == {"status": "ok"}

def test_version_returns_weights():
    response = client.get("/version")
    assert response.status_code == 200
    data = response.json()
    assert "weights" in data
    assert "detectors" in data

def test_version_weights_sum_to_one():
    response = client.get("/version")
    data = response.json()
    total = sum(data["weights"].values())
    assert abs(total - 1.0) < 0.001, f"Weights sum to {total}, not 1.0"

def test_file_too_large():
    # Real JPEG magic bytes + padding to exceed size limit
    jpeg_header = bytes([0xFF, 0xD8, 0xFF, 0xE0])
    large_content = jpeg_header + b"0" * (21 * 1024 * 1024)
    response = client.post(
        "/detect",
        files={"file": ("large.jpg", large_content, "image/jpeg")}
    )
    assert response.status_code == 413

def test_invalid_file_type():
    response = client.post("/detect", files={"file": ("test.txt", b"hello", "text/plain")})
    assert response.status_code == 400

def test_missing_file():
    response = client.post("/detect")
    assert response.status_code == 422  # FastAPI validation error

def test_authentic_photo():
    fixture_path = os.path.join(FIXTURE_DIR, "authentic.jpg")
    if not os.path.exists(fixture_path):
        pytest.skip("Fixture authentic.jpg not found")
    with open(fixture_path, "rb") as f:
        response = client.post("/detect", files={"file": ("authentic.jpg", f, "image/jpeg")})
    assert response.status_code == 200
    data = response.json()
    assert data["verdict"] in ["RAW", "Uncertain", "AI"]
    assert 0 <= data["confidence"] <= 100
    assert "details" in data
    expected_keys = [
        "c2pa", "forensic", "ela", "frequency", "color_entropy",
        "metadata", "copy_move", "jpeg_ghost", "cfa", "lighting",
        "noise_inconsistency", "grre", "color_correlation",
        "sharpness", "xmp"
    ]
    for key in expected_keys:
        assert key in data["details"], f"Missing detector: {key}"
